76 template<class T, int rank>
77 blitz::Array<T, rank> c_to_f(blitz::Array<T, rank> &arr)
78 {
79  // Initialize an 11-dim vector of transpositions
80  // (because transpose() doesn't take a TinyVector)
81  int const max_dims = 11;
82  int rev[max_dims];
83  for (int i=rank; i<max_dims; ++i) rev[i] = 0;
84 
85  // Reverse dimensions
86  for (int i=0; i<rank; ++i) rev[i] = rank-i-1;
87  auto ret(arr.transpose(rev[0], rev[1], rev[2], rev[3], rev[4], rev[5], rev[6], rev[7], rev[8], rev[9], rev[10]));
88 
89  // Re-base to 1
90  blitz::TinyVector<int, rank> base(1);
91  ret.reindexSelf(base);
92 
93  return ret;
94 }

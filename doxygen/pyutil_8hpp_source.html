109 template<class T, int N>
110 blitz::Array<T,N> py_to_blitz(
111 PyObject *ovec,
112 std::string const &vname,
113 int ndim, // # dimensions from dimension array (user-supplied)
114 int const *dims)
115 {
116  // Check data types and cast
117  PyArrayObject *vec = check_dimensions(ovec, vname, get_type_num<T>(), ndim, dims, N);
118 
119  int T_size = sizeof(T);
120  assert(T_size == PyArray_ITEMSIZE(vec));
121 
122  blitz::TinyVector<int,N> shape(0);
123  blitz::TinyVector<int,N> strides(0);
124  npy_intp *arr_dimensions = vec->dimensions;
125  npy_intp *arr_strides = vec->strides;
126 
127  // Set up shape and strides
128  for (int i=0;i<N;++i) {
129  shape[i] = arr_dimensions[i];
130  // Python/Numpy strides are in bytes, Blitz++ in sizeof(T) units.
131  strides[i] = arr_strides[i]/T_size;
132  }
133 //printf("py_to_blitz: vec->data = %p\n", vec->data);
134 
135  return blitz::Array<T,N>((T*) vec->data,shape,strides,
136  blitz::neverDeleteData);
137 }

149 template<class T>
150 std::vector<T> py_to_vector(PyObject *ovec, std::string const &vname, int ilen = -1)
151 {
152  int dims[1] = {ilen};
153 
154  auto blitz_obj(py_to_blitz<T,1>(ovec, vname, 1, dims));
155  int len = blitz_obj.extent(0);
156  std::vector<T> ret;
157  ret.reserve(len);
158  for (int i=0; i<len; ++i) ret.push_back(blitz_obj(i));
159  return ret;
160 }

163 template<class T>
164 PyObject *vector_to_py(std::vector<T> const &vec)
165 {
166  PyObject *vec_py = NULL;
167  try {
168  int n = vec.size();
169  int dims[] = {n};
170  PyObject *vec_py = PyArray_FromDims(1, dims, get_type_num<T>());
171 
172  // Wrap as blitz for convenience
173  auto vec_bl(py_to_blitz<T,1>(vec_py, "vec_py", 1, dims));
174 
175  // Copy from the vector to the Blitz array
176  for (int i=0; i<n; ++i) vec_bl(i) = vec[i];
177  return vec_py;
178  } catch(...) {
179  // De-allocated since we're not returning
180  if (vec_py) Py_DECREF(vec_py);
181  throw std::exception();
182  }
183 }

185 template<class T, int N>
186 PyObject *blitz_to_py(blitz::Array<T,N> const &array)
187 {
188  PyObject *array_py = NULL;
189  try {
190  int dims[N];
191  for (int i=0; i<N; ++i) dims[i] = array.extent(i);
192  PyObject *array_py = PyArray_FromDims(N, dims, get_type_num<T>());
193 
194  // Wrap as blitz for convenience
195  auto array_bl(py_to_blitz<T,N>(array_py, "array_py", 1, dims));
196 
197  // Copy from the arraytor to the Blitz array
198  array_bl = array;
199  return array_py;
200  } catch(...) {
201  // De-allocated since we're not returning
202  if (array_py) Py_DECREF(array_py);
203  throw std::exception();
204  }
